## `infra/github/src/main.tf`

```hcl
locals {
  is_production = lower(var.ENVIRONMENT) == "production"
}

# Repository settings
resource "github_repository" "main" {
  name                  = var.repository.name
  description           = var.repository.description
  visibility            = var.repository.visibility
  auto_init             = var.repository.auto_init
  has_issues            = var.repository.has_issues
  has_discussions       = var.repository.has_discussions
  has_projects          = var.repository.has_projects
  has_wiki              = var.repository.has_wiki
  is_template           = var.repository.is_template
  default_branch        = var.repository.default_branch

  # Merge strategy settings
  allow_auto_merge                    = var.repository.allow_auto_merge
  delete_branch_on_merge              = var.repository.delete_branch_on_merge
  squash_merge_commit_title           = var.repository.squash_merge_commit_title
  squash_merge_commit_message         = var.repository.squash_merge_commit_message
  allow_squash_merge                  = var.repository.allow_squash_merge
  allow_merge_commit                  = var.repository.allow_merge_commit
  allow_rebase_merge                  = var.repository.allow_rebase_merge

  # Advanced settings
  web_commit_signoff_required = local.is_production ? true : false

  lifecycle {
    ignore_changes = [
      auto_init,
      archived,
      topics,
      pages,
    ]
  }
}

# Branch protection rules
resource "github_branch_protection" "main" {
  for_each = { for bp in var.branch_protections : bp.branch_name => bp }

  repository_id = github_repository.main.node_id
  pattern       = each.value.branch_name

  dynamic "required_status_checks" {
    for_each = each.value.require_status_checks ? [1] : []
    content {
      strict   = lookup(each.value, "require_status_checks_strict", false)
      contexts = local.is_production ? ["PR Checks"] : []
    }
  }

  dynamic "required_pull_request_reviews" {
    for_each = each.value.require_pull_request_reviews ? [1] : []
    content {
      require_code_owner_reviews      = each.value.require_code_owner_reviews
      required_approving_review_count = each.value.required_review_count
    }
  }

  required_linear_history = each.value.require_linear_history

  enforce_admins = false

  allows_force_pushes             = false
  allows_deletions                = false
  require_conversation_resolution = true
}

# GitHub Actions Secrets
resource "github_actions_secret" "secrets" {
  for_each = nonsensitive(var.secrets)

  repository      = var.repository.name
  secret_name     = each.key
  plaintext_value = each.value
}

# GitHub Actions Variables
resource "github_actions_variable" "variables" {
  for_each = var.variables

  repository    = var.repository.name
  variable_name = each.key
  value         = each.value
}

# Issue labels
resource "github_issue_label" "labels" {
  for_each = { for label in var.labels : label.name => label }

  repository  = var.repository.name
  name        = each.value.name
  description = each.value.description
  color       = trimprefix(each.value.color, "#")
}

# Repository topics
resource "github_repository_topics" "main" {
  repository = var.repository.name
  topics     = var.topics
}

# Outputs
output "GITHUB_REPOSITORY_ID" {
  description = "The GitHub repository ID"
  value       = github_repository.main.node_id
  sensitive   = false
}

output "GITHUB_REPOSITORY_URL" {
  description = "The GitHub repository URL"
  value       = github_repository.main.html_url
  sensitive   = false
}

output "GITHUB_REPOSITORY_SSH_CLONE_URL" {
  description = "The GitHub repository SSH clone URL"
  value       = github_repository.main.ssh_clone_url
  sensitive   = false
}
```

## `infra/github/import.sh`

```bash
#!/bin/bash

# GitHub Repository Bootstrap Helper Script
# Creates the GitHub repository with GitHub CLI when needed,
# then imports it into Terraform state and applies configuration.

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
cd "$SCRIPT_DIR"

REPOSITORY_NAME="$(node -p "require('./src/config.json').repository.name")"
REPOSITORY_DESCRIPTION="$(node -p "require('./src/config.json').repository.description ?? ''")"
REPOSITORY_VISIBILITY="$(node -p "require('./src/config.json').repository.visibility ?? 'private'")"

resolve_github_token() {
    local candidate=""

    for candidate in "${GH_PAT_TOKEN:-}" "${GITHUB_TOKEN:-}"; do
        if [ -n "$candidate" ] && curl -fsS -H "Authorization: Bearer $candidate" https://api.github.com/user >/dev/null 2>&1; then
            printf '%s' "$candidate"
            return 0
        fi
    done

    if gh auth status >/dev/null 2>&1; then
        gh auth token
        return 0
    fi

    return 1
}

echo "🔧 GitHub Repository Terraform Bootstrap"
echo "========================================="
echo ""

# Prerequisites check
if ! command -v npx &> /dev/null; then
    echo "❌ npm is not installed. Please install Node.js first."
    exit 1
fi

if ! command -v gh &> /dev/null; then
    echo "❌ GitHub CLI is not installed. Please install gh first."
    exit 1
fi

if [ ! -f ".env" ]; then
    echo "⚠️  .env file not found in infra/github/"
    echo ""
    echo "To set up the GitHub token:"
    echo "1. Create a Personal Access Token at https://github.com/settings/tokens/new"
    echo "2. Create .env with:"
    echo ""
    echo "GITHUB_TOKEN=ghp_xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx"
    echo "GITHUB_OWNER={{GITHUB_OWNER}}"
    echo ""
    echo "3. Run: npx dotenvx encrypt"
    echo "4. Re-run this script"
    exit 1
fi

if [ ! -f ".env.keys" ]; then
    echo "⚠️  .env.keys file not found"
    echo "Encrypting .env file..."
    npx dotenvx encrypt
fi

echo "✅ Environment configuration found"
echo ""

# Step 1: Initialize Terraform
echo "📦 Step 1: Initializing Terraform..."
GITHUB_AUTH_TOKEN="$(npx dotenvx run -- bash -c "$(declare -f resolve_github_token); resolve_github_token")"
export GITHUB_AUTH_TOKEN

if [ -z "$GITHUB_AUTH_TOKEN" ]; then
    echo "❌ Could not resolve a working GitHub token from .env or GitHub CLI"
    exit 1
fi

npx dotenvx run -- bash -c 'export TF_VAR_GITHUB_TOKEN="$GITHUB_AUTH_TOKEN"; export TF_VAR_GITHUB_OWNER="$GITHUB_OWNER"; cd src && terraform init'
echo "✅ Terraform initialized"
echo ""

# Step 2: Verify GitHub connection
echo "🔐 Step 2: Verifying GitHub connection..."
TOKEN_CHECK=$(npx dotenvx run -- bash -c 'TOKEN="$GITHUB_AUTH_TOKEN"; curl -s -H "Authorization: Bearer $TOKEN" https://api.github.com/user | grep -o "\"login\"" || echo "FAILED"')

if [ "$TOKEN_CHECK" = "FAILED" ]; then
    echo "❌ GitHub authentication failed"
    echo "Please check GITHUB_TOKEN in .env, or refresh GitHub CLI auth"
    exit 1
fi

echo "✅ GitHub authentication successful"
echo ""

# Step 3: Bootstrap repository state
echo "🚀 Step 3: Bootstrapping repository state..."
echo "This will either:"
echo "  - Create '${REPOSITORY_NAME}' in the configured GitHub organization if it does not exist"
echo "  - Or import '${REPOSITORY_NAME}' into Terraform state if it already exists"
echo "  - Then apply the Terraform configuration"
echo ""
read -p "Continue? (y/n) " -n 1 -r
echo
if [[ ! $REPLY =~ ^[Yy]$ ]]; then
    echo "Cancelled."
    exit 0
fi

echo ""
echo "Applying bootstrap workflow..."
npx dotenvx run -- bash -c '
set -euo pipefail
export TF_VAR_GITHUB_TOKEN="$GITHUB_AUTH_TOKEN"
export TF_VAR_GITHUB_OWNER="$GITHUB_OWNER"
REPOSITORY_NAME="$(node -p "require('\''./src/config.json'\'').repository.name")"
REPOSITORY_DESCRIPTION="$(node -p "require('\''./src/config.json'\'').repository.description ?? '\'''\''")"
REPOSITORY_VISIBILITY="$(node -p "require('\''./src/config.json'\'').repository.visibility ?? '\''private'\''")"

if curl -fsS -H "Authorization: Bearer $TF_VAR_GITHUB_TOKEN" "https://api.github.com/repos/$TF_VAR_GITHUB_OWNER/$REPOSITORY_NAME" >/dev/null; then
    echo "ℹ️  Repository already exists on GitHub"

    if ! cd src && terraform state show github_repository.main >/dev/null 2>&1; then
        echo "📥 Importing repository into Terraform state..."
        terraform import -var-file="config.json" github_repository.main "$REPOSITORY_NAME"
        cd ..
    else
        echo "ℹ️  Repository is already tracked in Terraform state"
        cd ..
    fi
else
    echo "🆕 Repository does not exist yet; creating with GitHub CLI"
    gh repo create "$TF_VAR_GITHUB_OWNER/$REPOSITORY_NAME" "--$REPOSITORY_VISIBILITY" --description "$REPOSITORY_DESCRIPTION" --add-readme

    echo "📥 Importing newly created repository into Terraform state..."
    cd src
    terraform import -var-file="config.json" github_repository.main "$REPOSITORY_NAME"
    cd ..
fi

cd src
terraform apply -auto-approve -var-file="config.json"
'

echo ""
echo "✅ Repository bootstrap completed!"
echo ""
echo "📋 Next steps:"
echo "  To see proposed changes:  cd infra/github/src && npx dotenvx run -- terraform plan -var-file=config.json"
echo "  To re-apply changes:      cd infra/github/src && npx dotenvx run -- terraform apply -var-file=config.json"
```

## `.github/workflows/github-infra-deploy.yml`

```yaml
name: Github Infrastructure Deploy

on:
  push:
    branches: [main]
    paths:
      - 'infra/github/**'
      - '.github/workflows/github-infra-deploy.yml'
  workflow_dispatch:

permissions:
  contents: read

jobs:
  check-changes:
    runs-on: ubuntu-latest
    name: Check for infra changes
    outputs:
      github-infra-changed: ${{ steps.check.outputs.github-infra-changed }}

    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 2

      - name: Check for changes in GitHub infra
        id: check
        run: |
          if [ "${{ github.event_name }}" = "workflow_dispatch" ]; then
            echo "github-infra-changed=true" >> $GITHUB_OUTPUT
            exit 0
          fi
          if git diff --name-only HEAD~1 -- infra/github/ .github/workflows/github-infra-deploy.yml | grep -q .; then
            echo "github-infra-changed=true" >> $GITHUB_OUTPUT
          else
            echo "github-infra-changed=false" >> $GITHUB_OUTPUT
          fi

  github-infra-deploy:
    needs: check-changes
    if: needs.check-changes.outputs.github-infra-changed == 'true'
    runs-on: ubuntu-latest
    name: GitHub Infra Deploy
    env:
      TF_VAR_ENVIRONMENT: production
      TF_VAR_GITHUB_OWNER: ${{ github.repository_owner }}
      DOTENV_PRIVATE_KEY: ${{ secrets.DOTENV_PRIVATE_KEY }}

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup pnpm
        uses: pnpm/action-setup@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "22"
          cache: "pnpm"

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Load secrets from encrypted .env
        run: |
          pnpm dotenvx run --env-file=infra/github/.env -- sh -c '
            echo "TF_VAR_GITHUB_TOKEN=$GH_PAT_TOKEN" >> "$GITHUB_ENV"
          '

      - name: Setup Terraform
        uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: latest
          terraform_wrapper: false

      - name: Terraform Init
        working-directory: infra/github/src
        run: terraform init

      - name: Bootstrap Terraform state from existing GitHub resources
        working-directory: infra/github/src
        run: |
          REPO_NAME=$(jq -r '.repository.name' config.json)
          if [ -z "$REPO_NAME" ] || [ "$REPO_NAME" = "null" ]; then
            echo "::error::Could not resolve repository.name from config.json"
            exit 1
          fi
          terraform import -var-file='config.json' github_repository.main "$REPO_NAME" || true
          terraform import -var-file='config.json' 'github_branch_protection.main["main"]' "$REPO_NAME:main" || true
          jq -r '.labels[].name' config.json | while read -r label; do
            [ -z "$label" ] && continue
            terraform import -var-file='config.json' "github_issue_label.labels[\"$label\"]" "$REPO_NAME:$label" || true
          done

      - name: Terraform Plan
        working-directory: infra/github/src
        run: terraform plan -var-file='config.json' -out=tfplan

      - name: Terraform Apply
        working-directory: infra/github/src
        run: terraform apply -auto-approve tfplan

      - name: Export Results
        working-directory: infra/github/src
        run: |
          echo "Repository ID: $(terraform output -raw GITHUB_REPOSITORY_ID)"
          echo "Repository URL: $(terraform output -raw GITHUB_REPOSITORY_URL)"

      - name: Post Deployment Summary
        run: |
          echo "### GitHub Infrastructure Deployed" >> $GITHUB_STEP_SUMMARY
          echo "Branch protection and repository settings applied." >> $GITHUB_STEP_SUMMARY
```

**Notes:**
- Requires `DOTENV_PRIVATE_KEY` repository secret (the value from `.env.keys`)
- The encrypted `.env` stores `GH_PAT_TOKEN` which becomes `TF_VAR_GITHUB_TOKEN`
- Triggers on push to `main` when `infra/github/` or the workflow itself changes
- Also supports manual trigger via `workflow_dispatch`
- Bootstraps state via `terraform import` (idempotent — `|| true` ignores already-imported resources)

## `infra/github/README.md`

Template:

```markdown
# GitHub Repository Infrastructure

Terraform configuration for managing the GitHub repository via Infrastructure as Code.

## Overview

The GitHub repository (`{{GITHUB_OWNER}}/{{REPO_NAME}}`) is managed through Terraform:
- Repository settings and metadata
- Branch protection rules
- Issue labels
- Security and analysis features
- GitHub Actions secrets and variables

## Prerequisites

- Node.js (for npx/dotenvx)
- GitHub CLI (`gh`)
- Terraform CLI
- Fine-grained Personal Access Token with permissions:
  - Administration (read & write)
  - Actions (read & write)
  - Contents (read only)
  - Issues (read & write)
  - Workflows (read & write)

## Quick start

1. Set your token in `.env` and encrypt:
   ```bash
   cd infra/github
   # Edit .env with your token
   npx dotenvx encrypt
   ```

2. Bootstrap the repository:
   ```bash
   bash import.sh
   ```

3. Make changes to `src/config.json` and apply:
   ```bash
   cd src
   npx dotenvx run -- terraform apply -var-file=config.json
   ```

## Files

| File | Purpose |
|------|---------|
| `.env` | Encrypted GitHub credentials |
| `.env.keys` | Decryption key (NEVER commit) |
| `import.sh` | Bootstrap script |
| `src/providers.tf` | GitHub provider config |
| `src/variables.tf` | Input variable definitions |
| `src/main.tf` | Repository resources |
| `src/config.json` | Configuration values |
```
